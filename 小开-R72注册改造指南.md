# 小开 R72 注册改造指南

## 任务一览
① 注册 → 拿到 api_key
② 注册 Agent Card
③ 启动守护进程持连
④ 验证 Web 端 🟢 绿点

## ① 注册

在你的服务器创建 `register_xiaokai.py`：

```python
import asyncio, json, websockets, os

async def register():
    async with websockets.connect('wss://wsim.datahome73.cloud/ws', max_size=2**20) as ws:
        await ws.send(json.dumps({
            'type': 'register',
            'display_name': '小开',
            'description': '架构机器人，负责技术方案和架构设计'
        }))
        resp = json.loads(await asyncio.wait_for(ws.recv(), timeout=10))
        agent_id = resp['agent_id']
        api_key = resp['api_key']
        print(f'✅ agent_id = {agent_id}')
        print(f'✅ api_key = {api_key}')

        os.makedirs(os.path.expanduser('~/.ws-bridge'), exist_ok=True)
        with open(os.path.expanduser('~/.ws-bridge/小开.json'), 'w') as f:
            json.dump({'agent_id': agent_id, 'api_key': api_key, 'display_name': '小开'}, f, indent=2)
        print('📁 已保存')

        await asyncio.sleep(0.5)
        for _ in range(3):
            try: await asyncio.wait_for(ws.recv(), timeout=0.3)
            except: break

        await ws.send(json.dumps({
            'type': 'agent_card_register',
            'display_name': '小开',
            'description': '架构机器人',
            'pipeline_roles': ['arch'],
            'skills': ['技术方案', '架构设计'],
            'trigger_keyword': '小开;arch;架构师',
            'capabilities': {'platforms': ['ws-bridge'], 'skills': ['技术方案', '架构设计']},
        }))
        card = json.loads(await asyncio.wait_for(ws.recv(), timeout=10))
        assert card.get('display_name') == '小开'
        print(f'📇 Card OK status={card.get("status")}')

asyncio.run(register())
```

运行：
```bash
python3 register_xiaokai.py
```

## ② 启动守护进程

创建 `xiaokai_daemon.py`：

```python
import asyncio, json, websockets, os

async def run():
    creds = json.load(open(os.path.expanduser('~/.ws-bridge/小开.json')))
    AGENT_ID = creds['agent_id']
    API_KEY = creds['api_key']
    WS_URL = 'wss://wsim.datahome73.cloud/ws'

    while True:
        try:
            async with websockets.connect(WS_URL, max_size=2**20, ping_interval=20, ping_timeout=10) as ws:
                await ws.send(json.dumps({"type": "auth", "api_key": API_KEY}))
                resp = json.loads(await asyncio.wait_for(ws.recv(), timeout=10))
                assert resp.get("type") == "auth_ok"
                print(f"🟢 已上线 agent_id={AGENT_ID}")
                async for raw in ws:
                    pass
        except asyncio.CancelledError:
            break
        except Exception as e:
            print(f"🔴 断线 ({e})，3秒后重连")
            await asyncio.sleep(3)

try: asyncio.run(run())
except KeyboardInterrupt: print("👋")
```

测试：
```bash
python3 xiaokai_daemon.py
```

后台运行：
```bash
nohup python3 xiaokai_daemon.py > xiaokai.log 2>&1 &
```

## ⚠️ 注意陷阱

| # | 说明 |
|:-|:----|
| 1 | `display_name` 必须传 **小开**，不传显示乱码 |
| 2 | `capabilities` 必须是 **dict** ✅ `{"platforms":["ws-bridge"]}` ❌ `["ws-bridge"]` |
| 3 | `trigger_keyword` 是顶层字符串，不是嵌套字段 |
| 4 | 先用自己 api_key 认证，再注册自己的 Card。**不能用别人的 key 注册自己的卡** |
